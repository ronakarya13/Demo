import { ChangeDetectorRef, OnInit } from "@angular/core";
import { animate, style, transition, trigger } from "@angular/animations";
import { Component } from "@angular/core";
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from "@angular/forms";
import { ApiProviderService } from "app/user-profile/service/api-provider.service";
import { Router } from "@angular/router";

@Component({
  selector: "app-pre-login",
  templateUrl: "./pre-login.component.html",
  styleUrls: ["./pre-login.component.scss"],
  animations: [
    trigger("visibilityChanged", [
      transition(":enter", [
        style({ opacity: 0, transform: "translateX(-40px)" }),
        animate(600, style({ opacity: 1, transform: "translateX(0)" })),
      ]),
      transition(":leave", [
        style({ opacity: 1, transform: "translateX(0)" }),
        animate(600, style({ opacity: 0, transform: "translateX(-40px)" })),
      ]),
    ]),
  ],
})
export class PreLoginComponent implements OnInit {
  User: any = [];
  myCondition = true;
  form: FormGroup = new FormGroup({
    username: new FormControl(""),
    password: new FormControl(""),
  });
  submitted = false;
  valueChangesSub: any;

  constructor(
    private formBuilder: FormBuilder,
    private _cdr: ChangeDetectorRef,
    private apiProvider: ApiProviderService,
    public router: Router
  ) {}

  ngOnInit(): void {
    this.form = this.formBuilder.group({
      username: ["", [Validators.required]],
      password: ["", [Validators.required]],
    });
  }

  get f(): { [key: string]: AbstractControl } {
    return this.form.controls;
  }

  onSubmit(): void {
    this.submitted = true;
    if (this.form.invalid) {
      return;
    }
    console.log(this.form.value);

    if (
      this.form.value.username.trim() != "" &&
      this.form.value.password.trim() != ""
    ) {
      this.apiProvider.getUsers().subscribe((res) => {
        console.log("users ", res);

        const user = res.find((a: any) => {
          return (
            a.username == this.form.value.username.trim() &&
            a.password == this.form.value.password.trim()
          );
        });
        if (user) {
          this.apiProvider.activeUsername = this.form.value.username.trim();
          sessionStorage.setItem('username',this.form.value.username.trim());
          this.router.navigate(["dashboard"]);
        } else {
          alert("Please verify the credentials");
        }
      });
    }
  }
  onReset(): void {
    this.submitted = false;
    this.form.reset();
  }
}
