import { HttpClient } from "@angular/common/http";
import { Component, OnInit } from "@angular/core";
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from "@angular/forms";
import { MatSnackBar } from "@angular/material/snack-bar";
import { ApiProviderService } from "app/user-profile/service/api-provider.service";

@Component({
  selector: "app-table-list",
  templateUrl: "./table-list.component.html",
  styleUrls: ["./table-list.component.css"],
})
export class TableListComponent implements OnInit {
  selected = new FormControl(0);
  users$: any;
  form: FormGroup;
  submitted = false;
  selectedUser: any;
  isEdit = false;
  constructor(private _snackBar: MatSnackBar,
    private formBuilder: FormBuilder,
    private apiProvider: ApiProviderService
  ) {
    this.getUsers();

    this.form = this.formBuilder.group({
      username: [{ value: "" }, [Validators.required]],
      password: ["", [Validators.required]],
      name: ["", [Validators.required]],
      email: ["", [Validators.required]],
      address: [""],
      city: [""],
      country: [""],
      pCode: [""],
      desc: [""],
    });
  }

  ngOnInit() {}
  getUsers() {
    this.apiProvider.getUsers().subscribe((data) => {
      this.users$ = data;
    });
  }
  editUser(a) {
    this.isEdit = true;
    this.form.patchValue({
      username: a.username,
      password: a.password,
      name: a.name,
      email: a.email,
      address: a.address,
      city: a.city,
      country: a.country,
      pCode: a.pCode,
      desc: a.desc,
    });
    this.selectedUser = a;
    this.tabSelection(1);
  }

  delete(user){
    let snackBarRef = this._snackBar.open('Delete User','Yes');
    snackBarRef.afterDismissed().subscribe(() => {
      this.deleteUser(user);
    });
  }
  deleteUser(user) {
    this.apiProvider.deleteUsers(user.id).subscribe((res) => {this.getUsers()
   this._snackBar.open('User Deleted Successfully','',{duration : 5000})
    });
  }

  onSubmit() {
    if (this.isEdit) {
      this.apiProvider
        .updateUsers(this.selectedUser.id, this.form.value)
        .subscribe((res: any) => {
          this._snackBar.open('User Updated Successfully','',{duration : 5000});
          this.selectedUser = res;
          this.getUsers();
        });
    } else {
      this.apiProvider.addUsers(this.form.value).subscribe((res: any) => {
        this._snackBar.open('User Added Successfully','',{duration : 5000});
        this.tabSelection(0);
        this.getUsers();
      });
    }
  }

  tabSelection(val){
    this.selected.setValue(val);
    if(val=0){
      this.isEdit = false;
      this.form.reset();
    }
  }
}
